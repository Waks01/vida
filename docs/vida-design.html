08 — Database Schema
PostgreSQL — Every Table
Complete schema with types and relationships. The user record now carries email, pin_hash, and a theme preference.
Table: users One row per account (email auth + PIN)
id UUID PK, gen_random_uuid()
email VARCHAR(255) UNIQUE NOT NULL
password_hash VARCHAR(255) NOT NULL
pin_hash VARCHAR(255) NULL — bcrypt of 4-6 digit PIN
display_name VARCHAR(100) NULL
coin_balance INTEGER NOT NULL DEFAULT 0
vip_until TIMESTAMPTZ NULL — active if > now
theme_preference VARCHAR(20) DEFAULT 'dark' — dark|light|cupcake|cyberpunk|sunset|valentine
is_creator BOOLEAN DEFAULT FALSE
is_admin BOOLEAN DEFAULT FALSE
created_at TIMESTAMPTZ DEFAULT NOW()
Table: series A drama series
id UUID PK
creator_id UUID FK → users(id)
title VARCHAR(255) NOT NULL
description TEXT NULL
genre_id INT FK → genres(id)
thumbnail_url TEXT NULL
status ENUM: 'pending','published','rejected'
total_views BIGINT DEFAULT 0
created_at TIMESTAMPTZ DEFAULT NOW()
Table: episodes One row per 1-min episode
id UUID PK
series_id UUID FK → series(id)
episode_number INT NOT NULL
title VARCHAR(255) NOT NULL
hls_url TEXT NULL
thumbnail_url TEXT NULL
duration_seconds INT NOT NULL
is_premium BOOLEAN DEFAULT TRUE — free first 5
coin_cost INT DEFAULT 25 — set via EPISODE_COIN_COST env (10 coins = ₦1)
status ENUM: 'ready','published','pending'
Table: coin_transactions Immutable double-entry ledger
id UUID PK
user_id UUID FK → users(id)
amount INTEGER — +credit / -debit
source ENUM: 'ad_watch','episode_unlock','daily_checkin','coin_purchase','subscription','referral','admin_adjust'
reference_id UUID — ad_view / purchase / episode
balance_after INTEGER — snapshot after txn
created_at TIMESTAMPTZ DEFAULT NOW()
Table: payments Provider charges (Paystack/Stripe/Google Pay)
id UUID PK
user_id UUID FK → users(id)
provider VARCHAR(20) — 'paystack'|'stripe'|'googlepay'
provider_ref VARCHAR(255) — external charge/order id
product_type VARCHAR(20) — 'coin_pack'|'subscription'
amount DECIMAL(10,2)
currency VARCHAR(3) DEFAULT 'USD'
status ENUM: 'pending','success','failed','refunded'
pin_verified BOOLEAN DEFAULT TRUE
created_at TIMESTAMPTZ DEFAULT NOW()
Table: ad_views Fraud + reward tracking
id UUID PK
user_id UUID FK → users(id)
ad_unit_id VARCHAR(255) NOT NULL
device_id VARCHAR(255) NOT NULL
coins_awarded INTEGER DEFAULT 0
completed_at TIMESTAMPTZ DEFAULT NOW()
Table: creators Creator profile + payout
id UUID PK
user_id UUID FK → users(id) UNIQUE
payout_method VARCHAR(50) — 'paystack','stripe','bank'
payout_details JSONB — account details
total_earnings DECIMAL(10,2) DEFAULT 0
status ENUM: 'pending','approved','rejected'
Table: subscriptions VIP state machine
id UUID PK
user_id UUID FK → users(id)
status ENUM: 'active','grace_period','expired','cancelled','refunded'
provider VARCHAR(20) — 'paystack'|'stripe'|'googlepay'
product_id VARCHAR(255)
starts_at TIMESTAMPTZ
expires_at TIMESTAMPTZ
cancelled_at TIMESTAMPTZ NULL
